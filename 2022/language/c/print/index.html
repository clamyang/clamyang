int main(void) {
    printf("Please input start letter for print\n");   
    char alpha;
    scanf("%c", &alpha);
    char A = 65;

    for (char t = A; t <= alpha; t++) {   
        for (char j = alpha; j > t; j--) {
            printf(" ");
        }
        
        for(char i = A; i < t; i++) {     
            printf("%c", i);
        }

            printf("%c", k);
        }
        
        printf("\n");
    }
    return 0;
}
